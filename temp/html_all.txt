  import { loadPartial, initSearchBar, fetchAndRenderHotels } from "/js/common.js";

  const params = new URLSearchParams(location.search);
  const city = params.get("city");
  const checkIn = params.get("checkIn");
  const checkOut = params.get("checkOut");
  const mapWrapper = document.getElementById("mapWrapper");
  const storedRooms = JSON.parse(sessionStorage.getItem("searchRooms") || '[{"adults":2,"children":0,"childAges":[]}]');

  initSearchBar(async ({ city, checkIn, checkOut, rooms }) => {
    const selectedRooms = rooms?.length ? rooms : storedRooms;
    sessionStorage.setItem("searchRooms", JSON.stringify(selectedRooms));
    const hasResult = await fetchAndRenderHotels(city, checkIn, checkOut, selectedRooms);
    mapWrapper.style.display = hasResult ? "block" : "none";

  <script>
    let mapInstance; // 전역에서 재사용

    function initMap() {
      const mapEl = document.getElementById("googleMap");
      if (!mapEl) return;

      // 기본 중심 위치 (도쿄 중심)
      const defaultCenter = { lat: 35.682839, lng: 139.759455 };

      // 이미 생성된 지도 있으면 재활용
      if (mapInstance) {
        mapInstance.setCenter(defaultCenter);
        return;
      }

      mapInstance = new google.maps.Map(mapEl, {
        center: defaultCenter,
        zoom: 12,
        disableDefaultUI: false,
      });

      // 예시용 마커
      new google.maps.Marker({
        position: defaultCenter,
        map: mapInstance,
        title: "도쿄 중심",
      });
    }

    // ✅ 모달 열릴 때 지도 초기화
    window.openMap = function () {
      const modal = document.getElementById("mapModal");
      modal.style.display = "flex";

      // DOM이 그려지고 나서 지도 생성
      setTimeout(() => {
        if (typeof google !== "undefined" && google.maps) {
          initMap();
        } else {
          console.error("Google Maps not ready.");
        }
      }, 200);
    };

    window.closeMap = function () {
      document.getElementById("mapModal").style.display = "none";
    };
  </script>

  <script>
    const city = "{{cityName}}";
    const BASE_PATH = "/";

    document.addEventListener("DOMContentLoaded", () => {
      const slug = window.citySlugMap?.[city] || city.toLowerCase();
      fetch(`${BASE_PATH}city-data/${slug}-top5.json`)
        .then(res => res.json())
        .then(data => {
          renderHotels(data.topRated, "hotelList-top");
          renderHotels(data.family, "hotelList-family");
          renderHotels(data.petFriendly, "hotelList-pet");
        })
        .catch(err => console.error("❌ 호텔 데이터 로드 실패:", err));

      document.querySelectorAll(".scroll-btn").forEach(btn => {
        btn.addEventListener("click", () => {
          const target = document.getElementById(btn.dataset.target);
          if (!target) return;
          target.scrollBy({
            left: btn.classList.contains("left")
              ? -target.clientWidth * 0.9
              : target.clientWidth * 0.9,
            behavior: "smooth"
          });
        });
      });
    });

    function renderHotels(hotels, targetId) {
      const wrapper = document.getElementById(targetId);
      if (!wrapper) return;
      const el = wrapper.classList.contains("horizontal-scroll")
        ? wrapper
        : wrapper.querySelector(".horizontal-scroll");
      if (!el) return;

      if (!hotels?.length) {
        el.innerHTML = "<p style='text-align:center;color:#777;'>표시할 숙소가 없습니다.</p>";
        return;
      }

      const cards = hotels
        .map(
          (h) => `
          <div class="hotel-card">
            <img src="${h.image || "https://picsum.photos/seed/" + encodeURIComponent(h.name) + "/400/250"}" alt="${h.name}">
            <div class="hotel-info">
              <strong>${h.name}</strong>
              <p>📍 ${h.address || h.city}</p>
              <p>⭐ ${h.rating || "4.5"} / 5.0</p>
            </div>
          </div>`
        )
        .join("");

      el.innerHTML = cards;
    }
  </script>
